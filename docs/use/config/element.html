我们将构成 item 级别的基础表单组件称之为元件，元件是构成表单元素的最小单元。

# Input

const interface InputParams {
  style?: React.CSSProperties;
  value?: string; // input的value,这个值一般交由form接管，请勿直接设置。
  onChange?: (val: string,form: any,cb: (val: string) => never) => void; // 输入时触发的回调函数，一般用来设置
  name: string; // placeholder
  width?: number | string; // 宽度
  type?: string; // 类型，即input的type属性
  disabled?: boolean | (form: any)=> boolean; // 是否disable
  readonly?: boolean; // 是否只读
  allowClear?: boolean; // 是否允许一键清空value
  onKeyDown?: (event: React.KeyboardEvent<HTMLInputElement>) => void; // 键盘事件回调件数
  onBlur?: () => void; // 失去焦点回调函数
  onFocus?: () => void; // 获取焦点回调函数
  className?: string; // 自定义class
  error?: string; // 错误文字，不传则不显示 ,这个值一般由form接管，请勿直接设置
  size?: 'm' | 'l'; // 输入框尺寸
  suffix?: (value: string) => React.ReactNode; // 后缀内容
}

componentType: {
    type: 'input',
    params: InputParams,
}

# Select

import { SelectProps, SelectValue } from 'antd/es/select';
const interface SelectParams extends SelectProps<SelectValue> {
  onChange?: (val: string,form: any,cb: (val: string) => never) => void; // 输入时触发的回调函数，一般用来设置
  disabled?: boolean | (form: any)=> boolean; // 是否disable
}

componentType: {
    type: 'select',
    params: SelectParams,
}

# Switch

const interface SwitchParams {
    onChange?: (val: string,form: any,cb: (val: string) => never) => void; // 输入时触发的回调函数，一般用来设置
    disabled?: boolean | (form: any)=> boolean; // 是否disable
    label?:string;
    checked: boolean; // 当前是否选中
    loading?: boolean; // 是否在loading
    style?: React.CSSProperties; // 自定义样式
    checkedText?: React.ReactNode; // 选中时显示的文案
    uncheckedText?: React.ReactNode; // 未选中时现实的额文案
    checkedBackgroundColor?: string; // 选中时的背景色
    uncheckedBackgroundColor?: string; // 未选中时的背景色
}

componentType: {
    type: 'switch',
    params: SwitchParams,
}
