customRule 本质是@zenlayer/validation的使用

customRule: {
    DomainName: [  //对应的ItemConfig.id
        (val: string) => {
            if (!val) {
              throw '必填';
            }
            if (val !== 'haha') {
              throw '请输入haha';
            }
        },
    ],
}
