前提条件

项目中已正常使用 zen-ui

1.安装依赖

npm i @zenlayer/form-plus

2.在项目中引入

import { FormPlus, PageSubmit } from '@zenlayer/form-plus';

3.创建组件

<FormPlus
  config={domainConfig}
  submit={<PageSubmit onSubmit={() => {}} submitLabel="Add Domain" />}
/>

4.编写 config

const config={
  product: 'cdn_Domain',
    data: {
      viewModel:{
        fields:[
          {
            sortId: '0',
            type: 'group',
            id: 'regions',
            title: 'Region',
            children: [
              {
                type: 'item',
                id: 'zones',
                componentType: {
                  type: 'zones',
                  params: {
                    name: 'Enter a doman name',
                    width: 552,
                  },
                },
              },
            ],
          },
        ]
      }
      customRule:{}
      sourceData:{}
      constraint:{}
    }
}
